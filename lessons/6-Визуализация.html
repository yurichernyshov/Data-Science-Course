def f_plotter(p_plt,num):
    x1=np.random.randn(num)
    y1=np.random.randn(num)
    p_plt.scatter(x1,y1,color="b", marker="d")
    x2=np.random.rand(num)
    y2=np.random.rand(num)
    p_plt.scatter(x2,y2,color="r", marker="<")
    p_plt.show()
f_plotter(plt,100)
